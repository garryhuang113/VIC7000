data_dict={'a':1, 'b':2, 'c':3, 'd':4}

def GetParameter(**kwargs):
	
	global data_dict
	
	return_dict={}
	para=kwargs['para']
	para=para.split(', ')
	
	for i in range(len(para)):
		try:
			return_dict[para[i]]=data_dict[para[i]]
		except:
			if 	para[i]=='':
				return_dict['Error']='Empty Parameter'
			else:
				return_dict[para[i]]='Parameter Unknown'
	return return_dict
